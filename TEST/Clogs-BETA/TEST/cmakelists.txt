#--------
set(p_name CLOGS_TEST)
set(p_ver 1.0)
#--------

cmake_minimum_required(VERSION 3.10...3.30)

include(FetchContent)
FetchContent_Declare(CLOGS GIT_REPOSITORY https://github.com/SG123it/Clogs/tree/BETA)
FetchContent_MakeAvailable(CLOGS)

project(${p_name} VERSION ${p_ver})

set(SRC
    main.cpp
)

add_executable(${p_name} ${SRC})
target_link_libraries(${p_name} CLOGS)
